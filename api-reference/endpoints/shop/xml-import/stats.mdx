# Import Statistics

Get statistics about imported products and offers from XML feeds.

## Endpoint

```
GET /{locale}/shop/xml-import/stats
```

## Authentication

<ParamField header="X-API-Key" type="string" required>
  Partner API key
</ParamField>

<ParamField header="X-API-Secret" type="string" required>
  Partner API secret
</ParamField>

## Response

<ResponseField name="data" type="object">
  <Expandable title="import statistics">
    <ResponseField name="total_products" type="integer">
      Total unique products imported
    </ResponseField>
    
    <ResponseField name="total_offers" type="integer">
      Total product offers (variations)
    </ResponseField>
    
    <ResponseField name="products_with_images" type="integer">
      Products that have images
    </ResponseField>
    
    <ResponseField name="products_with_discount" type="integer">
      Products with discounted prices
    </ResponseField>
    
    <ResponseField name="products_in_stock" type="integer">
      Products currently in stock
    </ResponseField>
    
    <ResponseField name="categories_used" type="integer">
      Number of categories with products
    </ResponseField>
    
    <ResponseField name="last_import" type="string">
      Last import timestamp (ISO 8601)
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>

```bash cURL
curl -X GET "https://staging-api.loyalty.lt/en/shop/xml-import/stats" \
  -H "X-API-Key: your_api_key" \
  -H "X-API-Secret: your_api_secret"
```

</RequestExample>
