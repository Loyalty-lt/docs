# Sync Status

Get the current synchronization status for partner's offers and products.

## Endpoint

```
GET /{locale}/shop/sync-status
```

## Authentication

<ParamField header="X-API-Key" type="string" required>
  Partner API key
</ParamField>

<ParamField header="X-API-Secret" type="string" required>
  Partner API secret
</ParamField>

## Response

<ResponseField name="data" type="object">
  <Expandable title="sync status">
    <ResponseField name="total_offers" type="integer">
      Total number of offers
    </ResponseField>
    
    <ResponseField name="active_offers" type="integer">
      Number of active offers
    </ResponseField>
    
    <ResponseField name="expired_offers" type="integer">
      Number of expired offers
    </ResponseField>
    
    <ResponseField name="last_sync" type="string">
      Last synchronization timestamp (ISO 8601)
    </ResponseField>
  </Expandable>
</ResponseField>
