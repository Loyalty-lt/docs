# List Shops

Get a paginated list of shops (physical locations) belonging to the authenticated partner.

## Endpoint

```
GET /{locale}/shop/shops
```

## Authentication

<ParamField header="X-API-Key" type="string" required>
  Partner API key
</ParamField>

<ParamField header="X-API-Secret" type="string" required>
  Partner API secret
</ParamField>

## Query Parameters

<ParamField query="page" type="integer">
  Page number for pagination
  
  **Default:** `1`
</ParamField>

<ParamField query="per_page" type="integer">
  Items per page (max 50)
  
  **Default:** `15`
</ParamField>

<ParamField query="search" type="string">
  Search by shop name or address
</ParamField>

<ParamField query="is_active" type="boolean">
  Filter by active status
  
  **Default:** `true`
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of shop objects
  
  <Expandable title="shop object">
    <ResponseField name="id" type="integer">
      Shop unique identifier
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Shop name
    </ResponseField>
    
    <ResponseField name="address" type="string">
      Full address
    </ResponseField>
    
    <ResponseField name="city" type="string">
      City name
    </ResponseField>
    
    <ResponseField name="phone" type="string">
      Contact phone number
    </ResponseField>
    
    <ResponseField name="working_hours" type="object">
      Working hours by day
    </ResponseField>
    
    <ResponseField name="is_active" type="boolean">
      Whether shop is active
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Pagination metadata
</ResponseField>

