# List Offers

Get a paginated list of available offers and promotions for the authenticated user.

## Endpoint

```
GET /{locale}/shop/offers
```

## Authentication

Requires JWT Bearer token (user must be logged in).

## Query Parameters

<ParamField query="page" type="integer">
  Page number
  
  **Default:** `1`
</ParamField>

<ParamField query="per_page" type="integer">
  Items per page (max 50)
  
  **Default:** `15`
</ParamField>

<ParamField query="partner_id" type="integer">
  Filter by specific partner
</ParamField>

<ParamField query="category_id" type="integer">
  Filter by category
</ParamField>

<ParamField query="type" type="string">
  Filter by offer type: `discount`, `points_multiplier`, `free_item`, `cashback`
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of offer objects
  
  <Expandable title="offer object">
    <ResponseField name="id" type="integer">
      Offer ID
    </ResponseField>
    
    <ResponseField name="title" type="object">
      Localized title
    </ResponseField>
    
    <ResponseField name="description" type="object">
      Localized description
    </ResponseField>
    
    <ResponseField name="type" type="string">
      Offer type
    </ResponseField>
    
    <ResponseField name="discount_value" type="number">
      Discount amount or percentage
    </ResponseField>
    
    <ResponseField name="points_required" type="integer">
      Points needed to claim (if applicable)
    </ResponseField>
    
    <ResponseField name="valid_from" type="string">
      Start date (ISO 8601)
    </ResponseField>
    
    <ResponseField name="valid_until" type="string">
      End date (ISO 8601)
    </ResponseField>
    
    <ResponseField name="partner" type="object">
      Partner information
    </ResponseField>
  </Expandable>
</ResponseField>
