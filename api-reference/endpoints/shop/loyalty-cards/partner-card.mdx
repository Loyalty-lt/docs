# Get Loyalty Card

Get detailed information about a specific partner card template by ID.

## Endpoint

```
GET /{locale}/shop/loyalty-cards/{id}
```

## Authentication

<ParamField header="X-API-Key" type="string" required>
  Partner API key
</ParamField>

<ParamField header="X-API-Secret" type="string" required>
  Partner API secret
</ParamField>

## Path Parameters

<ParamField path="locale" type="string" required>
  Language code (e.g., `en`, `lt`)
</ParamField>

<ParamField path="id" type="integer" required>
  Partner card template ID
</ParamField>

## Response

<ResponseField name="data" type="object">
  <Expandable title="partner card object">
    <ResponseField name="id" type="integer">
      Card template ID
    </ResponseField>
    
    <ResponseField name="name" type="object">
      Localized card name (`lt`, `en`)
    </ResponseField>
    
    <ResponseField name="description" type="object">
      Localized description
    </ResponseField>
    
    <ResponseField name="points_per_euro" type="number">
      Points earned per euro spent
    </ResponseField>
    
    <ResponseField name="welcome_bonus" type="integer">
      Welcome bonus points for new cards
    </ResponseField>
    
    <ResponseField name="is_active" type="boolean">
      Whether template is active
    </ResponseField>
    
    <ResponseField name="card_design" type="object">
      Card visual design settings
    </ResponseField>
  </Expandable>
</ResponseField>
