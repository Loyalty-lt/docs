# Products Sync Status

Get detailed synchronization status for partner's products.

## Endpoint

```
GET /{locale}/shop/products/sync-status
```

## Authentication

<ParamField header="X-API-Key" type="string" required>
  Partner API key
</ParamField>

<ParamField header="X-API-Secret" type="string" required>
  Partner API secret
</ParamField>

## Response

<ResponseField name="data" type="object">
  <Expandable title="sync status">
    <ResponseField name="total_products" type="integer">
      Total unique products
    </ResponseField>
    
    <ResponseField name="active_products" type="integer">
      Currently active products
    </ResponseField>
    
    <ResponseField name="inactive_products" type="integer">
      Inactive/disabled products
    </ResponseField>
    
    <ResponseField name="products_with_discount" type="integer">
      Products with active discounts
    </ResponseField>
    
    <ResponseField name="products_in_stock" type="integer">
      Products available in stock
    </ResponseField>
    
    <ResponseField name="last_sync" type="string">
      Last synchronization timestamp
    </ResponseField>
    
    <ResponseField name="categories_used" type="integer">
      Number of categories with products
    </ResponseField>
  </Expandable>
</ResponseField>
